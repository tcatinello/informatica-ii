Diagrama de estados (STD) — el estado activo procesa el próximo byte

Código recibir() — el case activo se resalta

switch (siguienteEstado) {
  case EST_STX:
    if (datos[i] == STX) {        // 0x02
      indBuffer = 0; cantBuffer = 0; chkBuffer = 0;
      siguienteEstado = EST_CANT;
    }                                // si no es STX, sigue en EST_STX
    break;
  case EST_CANT:
    cantBuffer = datos[i];
    if (cantBuffer == 0)         siguienteEstado = EST_CHK;
    else                         siguienteEstado = EST_DATOS;
    break;
  case EST_DATOS:
    buffer[indBuffer] = datos[i]; indBuffer++;
    chkBuffer ^= datos[i]; cantBuffer--;
    if (cantBuffer == 0) siguienteEstado = EST_CHK;
    break;
  case EST_CHK:
    if (datos[i] == chkBuffer) siguienteEstado = EST_ETX;
    else                      siguienteEstado = EST_STX;
    break;
  case EST_ETX:
    if (datos[i] == ETX) mostrarPaquete(buffer, indBuffer);
    siguienteEstado = EST_STX;
    break;
}
